進行資料庫連線，並實施新增、查詢、修改、刪除等功能

1.App-設定程式進入點，並透過無窮迴圈、switch以執行所需功能

import java.sql.*;
public class App {
    public static void main(String[] args) {
        StockDAO sto=new StockDaoImpl();
        boolean exit=false;
        while(!exit){
            AppTool.displayMenu();
            int funcNum=MenuTool.funcNum();
            switch(funcNum){
                case 1:
                Stock newSto= AppTool.gatDetails();
                try{
                    sto.addStocks(newSto);
                    System.out.println("新增成功");
                }catch(SQLException e){
                    System.out.println("新增股票資訊時發生錯誤"+e.getMessage());
                }
                break;

                case 2:
                Stock upadteSto= AppTool.gatDetails();
                try{
                    sto.updateStocks(upadteSto);
                    System.out.println("更新成功");
                }catch(SQLException e){
                    System.out.println("更新股票資訊時發生錯誤"+e.getMessage());
                }
                break;

                case 3:
                int delNumber=AppTool.getNumber();
                try{
                    sto.deleteStocks(delNumber);
                    System.out.println("刪除成功");
                }catch(SQLException e){
                    System.out.println("刪除股票資訊時發生錯誤"+e.getMessage());
                }
                break;

                case 4:
                int noStr=AppTool.getNumber();
                try{
                    Stock stoc=sto.getStocksById(noStr);
                    if(stoc != null){
                        System.out.println("股票代碼: "+stoc.getNumber());
                        System.out.println("股票名稱: "+stoc.getName());
                        System.out.println("股價: "+stoc.getStock_price());
                        System.out.println("營收: "+stoc.getRevenue());
                    }
                }catch(SQLException e){
                    System.out.println("查詢股票資訊時發生錯誤"+e.getMessage());
                }
                break;
                
                case 5:
                VisibleStockFrame frame=new VisibleStockFrame();
                frame.setVisible(true);
                break;

                case 6:
                System.out.println("謝謝使用");
                exit=true;
                break;

                default:
                System.out.println("輸入錯誤，請重新輸入");
                break;
            }
        }
    }
}

    }
}

3.Stock-設立建構式setter/getter

public class Stock{
    private int number;
    private String name;
    private double stock_price;
    private int revenue;

    public Stock(int arg_number, String arg_name, double arg_stock_price, int arg_revenue){
        this.number = arg_number;
        this.name = arg_name;
        this.stock_price = arg_stock_price;
        this.revenue = arg_revenue;
    }

    public int getNumber(){
        return this.number;
    }
    public void setNumber(int arg_number){
        this.number = arg_number;
    }

    public String getName(){
        return this.name;
    }
    public void setName(String arg_name){
        this.name = arg_name;
    }

    public double getStock_price(){
        return this.stock_price;
    }
    public void setStock_price(double arg_stock_price){
        this.stock_price = arg_stock_price;
    }

    public int getRevenue(){
        return this.revenue;
    }
    public void setRevenue(int arg_revenue){
        this.revenue = arg_revenue;
    }

}

4.StockDAO-設立抽象interface，包含新增、修改、查詢、刪除功能

import java.sql.SQLException;
import java.util.List;
public interface StockDAO {
    void addStocks(Stock stock) throws SQLException;
    void updateStocks(Stock stock) throws SQLException;
    void deleteStocks(int arg_number) throws SQLException;
    Stock getStocksById(int arg_number) throws SQLException;
}

5.StockDaoImpl-實現抽象StockDAO的功能

import java.sql.*;
import java.util.ArrayList;
import java.util.List;
public class StockDaoImpl implements StockDAO {
    private Connection conn;
    //建構子
    public StockDaoImpl(){
        //取得資料庫連線
        this.conn=DatabaseConnection.getConnection();
    }
    @Override
    /**
     * 新增股票資料
     */
    public void addStocks(Stock stock) throws SQLException{
        String sqlStr="insert into stock(number, name, stock_price, revenue) values (?, ?, ?, ?);";
        try(PreparedStatement st=conn.prepareStatement(sqlStr)){
            st.setInt(1, stock.getNumber());
            st.setString(2, stock.getName());
            st.setDouble(3, stock.getStock_price());
            st.setInt(4, stock.getRevenue());
            st.executeUpdate();
        }
    }

    @Override
    /**
     * 更新股票資料
     */
    public void updateStocks(Stock stock) throws SQLException{
        String sqlStr="update stock set name=?, stock_price=?, revenue=? where number=?;";
        try(PreparedStatement st=conn.prepareStatement(sqlStr)){
            st.setString(1, stock.getName());
            st.setDouble(2, stock.getStock_price());
            st.setInt(3, stock.getRevenue());
            st.setInt(4, stock.getNumber());
            st.executeUpdate();
        }
    }

    @Override
    /**
     * 刪除股票資料
     */
    public void deleteStocks(int arg_number) throws SQLException{
        String sqlStr="delete from stock where number=?;";
        try(PreparedStatement st=conn.prepareStatement(sqlStr)){
            st.setInt(1, arg_number);           
            st.executeUpdate();
         }
    }

    @Override
    /**
     * 利用股票代號查詢股票資料
     */
    public Stock getStocksById(int arg_number) throws SQLException{
        String sqlStr="select * from stock where number=? ;";
        try(PreparedStatement st=conn.prepareStatement(sqlStr)){
            st.setInt(1, arg_number);           
            ResultSet rs=st.executeQuery();
            if(rs.next()){
                int tmpNumber=rs.getInt("number");
                String tmpName=rs.getString("name");
                Double tmpStock_price=rs.getDouble("stock_price");
                int tmpRevenue=rs.getInt("revenue");
                return new Stock(tmpNumber, tmpName, tmpStock_price, tmpRevenue);
            }
        }
    return null;
    }
}
